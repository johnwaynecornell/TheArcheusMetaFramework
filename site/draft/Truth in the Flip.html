    bool next = rand.Next(2) == 1;

    // If guessing change, expect !last. If guessing same, expect last.
    if ((guess ? (!last) : last) == next)
        guessed++;

    // Update guess for next round based on change observed this round
    guess = next == last;

    if (next == last)
        same++;
    else
        different++;

    last = next;
}

Console.WriteLine(
